# Javascript for Rubist

# 1. 基本元素: 輸入、計算、輸出

  • 標準輸出

Ruby

puts 'hello world!'
# hello world

Javascript

console.log('hello world!')
// hello world
  • 變數宣告
    • ruby 可不宣告自動辨識型態
    • javascript 則建議要使用 var、let、const 來做宣告

Ruby

a = 1
a = '123'

Javascript

// var 宣告全域變數
var a = 1;
var a = '123'; //宣告幾次沒事

// let 宣告區域內變數
let b = 1;
let b = 2; // 宣告過的不能宣告第二次
// Uncaught SyntaxError: Identifier 'a' has already been declared

// const 宣告區域內常數
const c = 3.14;
c = 3.3; // 常數不能改值
//Uncaught TypeError: Assignment to constant variable.

//字串內變數
console.log(`hello ${a}`)
// hello 123
  • 字串內變數
    • ruby "#{x}"
    • javascript `${x}`

Ruby

puts "hello #{a}"
# hello 123

Javascript

console.log(`hello ${a}`)
// hello 123
  • 資料型態
    • Ruby: TrueClass/FalseClass、Integer、Float、String、Array、Hash、Symbol 等
    • Javascript: boolean、number、string、object、function 等
  • 資料類型判斷方式
    • typeof xxx

Ruby

1.class
# => Integer
1.0.class
# => Float
'hello'.class
# => String
[1,2].class
# => Array
{a: 1, b: 2}.class
# => Hash
nil.class
# => NilClass
true.class
# => TrueClass
false.class
# => FalseClass
:hello.class
# => Symbol

Javascript

typeof 1
// "number"
typeof 1.0
// "number"
typeof 'hello'
// "string"
typeof [1,2]
// "object"
typeof {a: 1, b: 2}
// "object"
typeof null
// "object"
typeof true
// 'boolean'
typeof false
// 'boolean'
typeof function(){}
// 'function'
  • 資料型態轉換
    • 數字轉字串
      • ruby 因為數字本身就是 object 所以可以直接用 instance_method 來轉換
      • javascript 的數字則需要先宣告再來轉換

Ruby

# [數字 -> 字串]
1.to_s # => "1"


# [字串 -> 整數]
'1'.to_i # => 1

# [字串 -> 浮點數]
'1.1'.to_f # => 1.1

# [無條件捨去 / 無條件進位 / 四捨五入 ]
10.555.floor # => 10
10.555.ceil # => 11
10.555.round # => 11
10.555.round(1) # => 10.6
10.555.round(2) # => 10.56

Javascript

// [數字 -> 字串]
const num = 1
num.toString() // '1'

// [字串 -> 整數]
parseInt('1') // 1

// [字串 -> 浮點數]
parseFloat('1.1') // 1.1

// [無條件捨去 / 無條件進位 / 四捨五入 ]
Math.floor(10.555) // 10
Math.ceil(10.555) // 11
Math.round(10.555) // 11
Math.round(10.555 * 10) / 10 // 10.6
Math.round(10.555 * 100) / 100 // 10.56

# 2. 控制流程

  • Conditional Control 條件控制
    • 布林 Boolean 資料型別: true 或 false
    • 關係運算子 == != < > <= >=
    • 邏輯運算子 && || !
  • Control Flow 控制流程
    • if else

Ruby

# [if-else]
if 1 > 2
  puts 'A'
elsif 3 >= 5
  puts 'B'
else
  puts 'C'
end

# [case-when]
a = 3
case a
when 1
  puts 'A'
when 3..5, 7
  puts 'B'  
else
  puts 'C'
end
# B

Javascript

// [if-else]
if (1 > 2) {
  console.log('A')
} else if (3 >= 5) {
  console.log('B')
} else {
  console.log('C')
}

// [switch-case]
const a = 3
switch (a) {
  case 1:
    console.log('A')
    break
  case 3:
  case 4:
  case 5:
  case 7:
    console.log('B')
    break
  default:
    console.log('C')
}
// B

# 4. 資料結構: 陣列 Array

  • ruby 和 javascript 在 Array 使用上幾乎一樣

Ruby

# [Array]
array = ['a', 'b', 'c']
array.each do |item|
  puts item
end
# a
# b
# c

# [Array-Hash]
ary_hash = [{a: 1, b: 2}, {a: 3, b: 4}]
ary_hash.each do |item|
  puts "#{item[:a]} | #{item[:b]}"
end
# 1 | 2
# 3 | 4

Javascript

// [Array]
const array = ['a', 'b', 'c']
for (const item of array){
    console.log(item)
}
// a
// b
// c

// [Array-Object]
const aryObj = [{a: 1, b: 2}, {a: 3, b: 4}]
for (const item of aryObj){
    console.log(`${item['a']} | ${item['b']}`)
}
// 1 | 2
// 3 | 4

# 5. 資料結構: 雜湊 Hash

  • 名稱不同
    • ruby 是 hash
    • javascript 相當於 object
  • 一般的 object 要加 Object.entries() 才能放到 for if 裡使用
    • Object.entries() 就是把 object 轉成 array-object
    • 相當於 ruby 用 .to_a 把 hash 轉成 array-hash

Ruby

# [Hash]
hash = {a: 1, b: 2}
hash.each do |key, val|
  puts "#{key} = #{val}"
end
# a = 1
# b = 2

# [Array-Hash]
ary_hash = [{a: 1, b: 2}, {a: 3, b: 4}]
ary_hash.each do |item|
  puts "#{item[:a]} | #{item[:b]}"
end
# 1 | 2
# 3 | 4

Javascript

// [Object] 
const obj = {a: 1, b: 2}
for (const [key, val] of Object.entries(obj)){
    console.log(`${key} = ${val}`)
}
// a = 1
// b = 2

// [Array-Object]
const aryObj = [{a: 1, b: 2}, {a: 3, b: 4}]
for (const item of aryObj){
    console.log(`${item['a']} | ${item['b']}`)
}
// 1 | 2
// 3 | 4

# 7. 資料結構: 物件 Object

Ruby

class Person
  def initialize(name)
    @name = name
  end

  def getFrom
    state = 'Taiwan'
    `${this.name} from ${state}.`
  end
end

john = Person.new('John')

puts john
puts john.state
puts john.getFrom

Javascript

class Person {
  constructor(name) {
    this.name = name;
  }
  
  getFrom() {
    const state = 'Taiwan';
    return `${this.name} from ${state}.`;
  }
}

const john = new Person('John');

console.log(JSON.stringify(john)); // { name: 'John' }  public 方法不再顯露於物件裡
console.log(JSON.stringify(john.state)); // undefined
console.log(JSON.stringify(john.getFrom())); // John from Taiwan.

# 8. 高階函式

接下來提到的高階函式，都使用以下宣告的 array 來講解

ruby

array = [
  {a: 1, b: 2},
  {a: 1, b: 3}
]

javascript

const array = [
  {a: 1, b: 2},
  {a: 1, b: 3}
]

# select vs filter()

找出 Array 中符合條件的 資料集合

  • 有兩種寫法
    • 縮寫
    • 不縮寫
  • 習慣
    • 通常 ruby 已縮寫寫法為主，若是長一點才不縮寫
    • javascript 目前也大致採用箭頭函式縮寫為主的寫法

ruby

# [不縮寫]
array.select do |x|
  x[:a]==1
end
# => [{:a=>1, :b=>2}, {:a=>1, :b=>3}]

# [縮寫]
array.select{|x| x[:a]==1}
# => [{:a=>1, :b=>2}, {:a=>1, :b=>3}]

javascript

// [不縮寫]
array.filter(function (x) {
  return x.a==1
})
//[ { a: 1, b: 2 }, { a: 1, b: 3 } ]

// [箭頭函式縮寫]
array.filter(x => x.a==1)
// [ { a: 1, b: 2 }, { a: 1, b: 3 } ]

# find vs find()

找出 Array 中符合條件的 首筆資料

ruby

# [不縮寫]
array.find do |x|
  x[:a]==1
end
# => {:a=>1, :b=>2}

# [縮寫]
array.find{|x| x[:a]==1}
# => {:a=>1, :b=>2}

javascript

// [不縮寫]
array.find(function (x) {
  return x.a==1
})
// { a: 1, b: 2 }

// [箭頭函式縮寫]
array.find(x => x.a==1)
// { a: 1, b: 2 }

# each vs forEach()

單純走訪所有資料，不回傳(return)

ruby

# [不縮寫]
array.each do |x|
  puts x[:a]
end
# 1
# 1

# [縮寫]
array.each{|x| puts x[:a]}
# 1
# 1

javascript

// [不縮寫]
array.forEach(function (x) {
  console.log(x.a)
})
// 1
// 1

// [箭頭函式縮寫]
array.forEach(x => console.log(x.a))
// 1
// 1

# map vs map()

走訪所有資料，並將回傳值組成一個新陣列

ruby

# [不縮寫]
array.map do |x|
  x[:a]
end
# => [1, 1]

# [縮寫]
array.map{|x| x[:a]}
# => [1, 1]

javascript

// [不縮寫]
array.map(function (x) {
  return x.a
})
// [ 1, 1 ]

// [箭頭函式縮寫]
array.map(x => x.a)
// [ 1, 1 ]

# all? vs every()

所有走訪的對象都符合條件的話，得到 true，否則 false

ruby

# [不縮寫]
array.all? do |x|
  x[:b]>2
end
# => false

# [縮寫]
array.all?{|x| x[:b]>2}
# => false

javascript

// [不縮寫]
array.every(function (x) {
  return x.b > 2
})
// false

// [箭頭函式縮寫]
array.every(x => x.b > 2)
// false

# any? vs some()

任一個走訪的對象符合條件的話，得到 true，否則 false

ruby

# [不縮寫]
array.any? do |x|
  x[:b]>2
end
# => true

# [縮寫]
array.any?{|x| x[:b]>2}
# => true

javascript

// [不縮寫]
array.some(function (x) {
  return x.b > 2
})
// true

// [箭頭函式縮寫]
array.some(x => x.b > 2)
// true

# reduce vs reduce()

可設定一個初始值，每次走訪會與前一個回傳的值再次作運算

ruby

# [不縮寫]
array.reduce(0) do |init, x|
  init + x[:a]
end
# => 2

# [縮寫]
array.reduce(0){|init, x| init+x[:a]}
# => 2

javascript

// [不縮寫]
array.reduce(function (init, x) {
  return init + x.a
}, 0)
// 2

// [箭頭函式縮寫]
array.reduce((init,x) => init + x.a, 0)
// 2

# 8. 其他部分: 箭頭函式

可使用箭頭函式替代原有的 fucntion() 寫法

// 原 function 寫法
const func = function (x) {
  return x + 1
}

// 改箭頭函式 （可將 function 改寫為 =>）
const func = (x) => { return x + 1 }

// 只有單行，可省略 return 和 {}
const func = (x) => x + 1

// 只有單個參數，也可省略 ()
const func = x => x + 1

# 箭頭函式跟和一般函式差異

最大的差異在於 this 所代表的意義

  • 一般函式: this 代表本身
  • 箭頭函式: this 代表函式上層本身

在箭頭函式還沒誕生的時代，每當需要用到上層的 this 時，都會先將 this 命另一個變數名稱(如：var that = this)，這樣在函式裡面能夠區分本身的 this 和上層的 that。

例如：以下範例，當按下這個元件時，自己的顏色會有紅綠藍的切換的效果（不過因為切換很快速，會直接停在藍色）

//[一般函式] - 需要使用 var that = that 來表示上層本身
var colours = ['red', 'green', 'blue']
document.getElementById('element').addEventListener('click', function() {
    // 如果想在下面forEach使用到這個按鈕
    var that = this
    colours.forEach(function(item) {
        //（Ｘ錯誤）
        // 因為 this 代表本身(item)，而非上層的 element
        this.style.backgroundColor = item
        //Uncaught TypeError: Cannot set property 'backgroundColor' of undefined

        //（Ｏ正確）
        // 因為 that 有宣告，代表上層的 this
        that.style.backgroundColor = item
    })
})

如果是寫了箭頭函式， this 會直接代表上層本身

//[箭頭函式] - 不需使用 var that = that 來表示上層本身
var colours = ['red', 'green', 'blue']
document.getElementById('element').addEventListener('click', function () {
    colours.forEach((item) => {
        //（Ｏ正確）
        // 因為 this 代表上層的本身(element)
        this.style.backgroundColor = item
    })
})

如果是寫了兩層的箭頭函式，this 會代表上上層本身，以此類推

//[多層箭頭函式] - this 會一直遞延下去
var colours = ['red', 'green', 'blue']
document.getElementById('element').addEventListener('click', function () {
    colours.forEach((item, index) => {
        setTimeout(() => {
          //（Ｏ正確）
          // 因為 this 代表上上層的本身(element)
          this.style.backgroundColor = item
        }, 1000 * index)
        // console.log(this)
    })
})

# 何種狀況不能使用箭頭函式

看到這邊就有一個疑問，那如果我寫了箭頭函式，但真的想用本身的 this，而非上層怎麼辦？

這時候只能換成一般函式的寫法。

以下說明幾種不能使用箭頭函式的狀況，實際上都和 this 有關。

  • 使用字面文字定義物件時，其中的方法
    const calculate = {
  array: [1, 2, 3],
  sum: () => {
    return this.array.reduce((result, item) => result + item)
  }
}

calculate.sum()
//TypeError: Cannot read property 'reduce' of undefined at Object.sum
  • 物件的prototype屬性中定義方法時
    function MyCat(name) {
  this.catName = name
}

MyCat.prototype.sayCatName = () => {
  return this.catName
}

cat = new MyCat('Mew')
cat.sayCatName()
// undefined
  • DOM事件處理的監聽者(事件處理函式)
    const button = document.getElementById('myButton')

button.addEventListener('click', () => {
  this.innerHTML = 'Clicked button'
})
//TypeError: Cannot set property 'innerHTML' of undefined
  • 建構函式
    const Message = (text) => {
  this.text = text
}

const helloMessage = new Message('Hello World!')
// Throws "TypeError: Message is not a constructor"

# 9. 其他部分: DOM 操作

DOM (Document Object Model) 是把 HTML 的標記文件(Document）轉換成 javascript 的物件(Object)。

JS 可以操作物件，但不能直接操作頁面上的標籤，所以需要透過 DOM 來操作 HTML 。

以下就來說明常見的 DOM 用法

# 取得 DOM 元素

// ES6 以前
document.getElementsByTagName('xxx')   // 取得 tag 是 'xxx' 的『元素集合』
document.getElementsByClassName('xxx') // 取得 class 有 'xxx' 的『元素集合』
document.getElementById('xxx')         // 取得 id 是 'xxx' 的『元素』

// ES6
document.querySelector('xxx .yyy #zzz')     // 選擇器方式選擇符合條件的『第一個元素』
document.querySelectorAll('xxx .yyy #zzz')  // 選擇器方式選擇符合條件的『元素集合』

# 改變元素 Style

//指定單一屬性
element.style.color = "blue"

//所有屬性變更
element.style.cssText = "color: blue; border: 1px solid black"
element.setAttribute("style", "color:red; border: 1px solid blue;")

# 改變元素 Class

const elt = document.querySelector('element')

//新增
element.classList.add('xxx')
element.classList.add('xxx','xxx2','xxx3')
//刪除
element.classList.remove('xxx')
element.classList.remove('xxx','xxx2','xxx3')
//相反
element.classList.toggle('xxx')
//是否包含
element.classList.contains('xxx')

# 改變元素的內容

element.innerText = 'buzz'
element.innerHTML = '<h2>buzz</h2>'
element.outerHTML = '<h2>buzz</h2>'

# 查詢/新增/插入/刪除元素

//查詢
element.children()   // 取得全部的子元素
element.firstChild() // 取得第一個子元素
element.lastChild()  // 取得最後一個子元素
element.parentNode() // 取得父元素

//新增
var newDiv = document.createElement("div") //新增元素
var newContent = document.createTextNode("Hi there and greetings!");  //文字內容

//插入
element.appendChild(newDiv)     //將元素內容新增到元素中
element.appendChild(newContent) //將文字內容新增到元素中

element.insertAdjacentHTML('beforebegin', text)  // 在 element 之前。
element.insertAdjacentHTML('afterbegin', text)   // 在 element 裡面，第一個子元素之前。
element.insertAdjacentHTML('beforeend', text)    // 在 element 裡面，最後一個子元素之後。
element.insertAdjacentHTML('afterend', text)     // 在 element 之後。

//刪除
element.removeChild(newDiv)     //將部分元素內容刪除
